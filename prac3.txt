#Perform Spam Classifier
import pandas as pdfrom sklearn.model_selection import train_test_split from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.naive_bayes import MultinomialNB

from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score df = pd.read_csv('spam.csv', encoding='latin-1')
df = df[['v1', 'v2']]

df.columns = ['label', 'text']
 
df['label'] = df['label'].map({'ham': 0, 'spam': 1})

X_train, X_test, y_train, y_test = train_test_split(df['text'], df['label'], test_size=0.2, random_state=42) vectorizer = TfidfVectorizer(stop_words='english', max_df=0.7)
X_train_tfidf = vectorizer.fit_transform(X_train) X_test_tfidf = vectorizer.transform(X_test)
clf = MultinomialNB() clf.fit(X_train_tfidf, y_train) y_pred = clf.predict(X_test_tfidf)
accuracy = accuracy_score(y_test, y_pred) precision = precision_score(y_test, y_pred) recall = recall_score(y_test, y_pred)
f1 = f1_score(y_test, y_pred) print(f'Accuracy: {accuracy:.4f}') print(f'Precision: {precision:.4f}') print(f'Recall: {recall:.4f}')
print(f'F1 Score: {f1:.4f}')
